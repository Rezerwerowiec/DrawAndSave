46       public fun save() {
47           val view : LinearLayout = findViewById<LinearLayout>(R.id.paint_inc)
48           val bitmap = Bitmap.createBitmap(view.width, view.height, Bitmap.Config.ARGB_8888)
49           val canvas = Canvas(bitmap)
50           view.draw(canvas)
51           val location : String = this.externalCacheDir!!.absolutePath.toString() + "/sign" + (1000..9999).random() + ".png"
52           val file = File(location)
53           try {
54               ActivityCompat.requestPermissions(this@MainActivity, arrayOf(Manifest.permission.WRITE_EXTERNAL_STORAGE), 123)
55                   if(checkSelfPermission(Manifest.permission.WRITE_EXTERNAL_STORAGE)
56                       == PackageManager.PERMISSION_GRANTED) {
57                       val saved = bitmap.compress(Bitmap.CompressFormat.PNG, 100, FileOutputStream(file))
58                       if(saved)
59                       Toast.makeText(baseContext, "Zapis udany, lokacja: " + location, Toast.LENGTH_LONG)
60                           .show()
61                   }
62           } catch (e: Exception) {
63               e.printStackTrace()
64               Toast.makeText(baseContext, "Zapis sie nie udal:" + e.toString(), Toast.LENGTH_LONG).show()
65   
66           }
