3    import android.annotation.SuppressLint
4    import android.content.Context
5    import android.graphics.Canvas
6    import android.graphics.Color
7    import android.graphics.Paint
8    import android.graphics.Path
9    import android.os.Build
10   import android.util.AttributeSet
11   import android.view.MotionEvent
12   import android.view.View
13   import android.view.ViewGroup
14   import androidx.annotation.RequiresApi
15   import pfhb.damian.drawandsave.MainActivity.Companion.paintBrush
16   import pfhb.damian.drawandsave.MainActivity.Companion.path
17   
18   class PaintView : View {
19   
20       var params : ViewGroup.LayoutParams? = null
21   
22       companion object{
23           var pathList = ArrayList<Path>()
24           var colorList = ArrayList<Int>()
25           var currentBrush = Color.BLUE
26       }
27       constructor(context: Context) : this(context, null){
28           init()
29       }
30       constructor(context: Context, attrs: AttributeSet?) : this(context, attrs, 0){
31           init()
32       }
33       constructor(context: Context, attrs: AttributeSet?, defStyleAttr: Int) : super(context, attrs, defStyleAttr) {
34           init()
35       }
36   
37       private fun init(){
38           paintBrush.isAntiAlias = true
39           paintBrush.color = currentBrush
40           paintBrush.style = Paint.Style.STROKE
41           paintBrush.strokeJoin = Paint.Join.ROUND
42           paintBrush.strokeWidth = 4f
43           params = ViewGroup.LayoutParams(ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.WRAP_CONTENT)
44       }
45   
46       @RequiresApi(Build.VERSION_CODES.R)
47       @SuppressLint("ClickableViewAccessibility")
48       override fun onTouchEvent(event: MotionEvent?): Boolean {
49           val x = event!!.x
50           val y = event.y
51   
52           when(event.action){
53               MotionEvent.ACTION_DOWN -> {
54                   path.moveTo(x,y)
55                   return true
56               }
57               MotionEvent.ACTION_MOVE -> {
58                   path.lineTo(x,y)
59                   pathList.add(path)
60                   colorList.add(currentBrush)
61               }
62               else -> return false
63           }
64           postInvalidate()
65           return false
66       }
67   
68       override fun onDraw(canvas: Canvas) {
69           for(i in pathList.indices){
70               paintBrush.color = colorList[i]
71               canvas.drawPath(pathList[i], paintBrush)
72               invalidate()
73           }
74       }
75   }
